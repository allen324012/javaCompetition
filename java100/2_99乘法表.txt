public class MainClass {
    public static void main(String[] args) {
        for (int i = 1; i < 10; i++) {
            for (int k = 1; k < 10; k++) {
                int n = i * k;
                System.out.print(i + " * " + k + " = "+ n + "\t");
            }
            System.out.println();
        }
    }
}
